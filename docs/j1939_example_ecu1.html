ECU demo

This example simulates a simple ECU. The following PGNs are used by this ECU example:

PGN Parameter Group Label Acronym Repetition Rate
0 Torque/Speed Control 1 TSC1 control purpose
61440 Electronic Retarder Controller 1 ERC1 100 ms
61443 Electronic Engine Controller 2 EEC2 10 ms
61444 Electronic Engine Controller 1 EEC1 10 ms
65240 Commanded Address CA -
65242 Software Identification SOFT on request
65247 Electronic Engine Controller 3 EEC3 250 ms
65251 Engine Configuration 1 EC1 on request
65253 Engine Hours, Revolutions HOURS on request
65260 Vehicle Identification VI on request
65262 Engine Temperature 1 ET1 1000 ms
65263 Engine Fluid Level/Pressure 1 EFL/P1 500 ms
65265 Cruise Control/Vehicle Speed 1 CCVS1 100 ms
65266 Fuel Economy (Liquid) LFE1 100 ms
65269 Ambient Conditions AMB 1000 ms
65270 Intake/Exhaust Conditions 1 IC1 500 ms
65271 Vehicle Electrical Power 1 VEP1 1000 ms

The implementation is done in the file j1939_demo_ecu1.c.

//--------------------------------------------------------------------------------------------------------------------//
// ECU_Pgn00000() //
// Torque / Speed Control 1 //
//--------------------------------------------------------------------------------------------------------------------//
int16_t ECU_Pgn00000(uint8_t * pubDataV, uint16_t CPP_PARM_UNUSED(uwSizeV), uint8_t ubSrcAddrV,
uint8_t CPP_PARM_UNUSED(ubNetV))
{
uint16_t uwSpeedT;
//---------------------------------------------------------------------------------------------------
// copy received data
//
tsECU_ControlS.ubDevice = ubSrcAddrV;
tsECU_ControlS.ubMode = *(pubDataV + 0);
uwSpeedT = *(pubDataV + 2);
uwSpeedT = uwSpeedT << 8;
uwSpeedT = uwSpeedT + *(pubDataV + 1);
tsECU_ControlS.uwSpeed = uwSpeedT;
tsECU_ControlS.ubTorque = *(pubDataV + 3);
return eJ1939_HANDLER_RESULT_OK;
}

//--------------------------------------------------------------------------------------------------------------------//
// ECU_Pgn61444() //
// EEC1 //
//--------------------------------------------------------------------------------------------------------------------//
int16_t ECU_Pgn61444(uint8_t * pubDataV, uint16_t CPP_PARM_UNUSED(uwSizeV), uint8_t CPP_PARM_UNUSED(ubSrcAddrV),
uint8_t CPP_PARM_UNUSED(ubNetV))
{
static uint16_t uwSpeedS = 200 * 8; // 1 Bit = 0.125rpm
static uint8_t ubTorqueS = 125; // 1 Bit = 1%, -125% Offset
static uint8_t ubSimDirS = 1; // direction of simulation
//---------------------------------------------------------------------------------------------------
// simulate torque
//
ubTorqueS = ubTorqueS + 1;
if (ubTorqueS > 200)
{
ubTorqueS = 125;
}
//---------------------------------------------------------------------------------------------------
// simulate engine speed in rpm between 200 and 2200
//
if (tsECU_ControlS.uwSpeed == 0)
{
if (ubSimDirS == 1)
{
uwSpeedS = uwSpeedS + (1 * 8);
if (uwSpeedS / 8 > 2200)
{
ubSimDirS = 0;
}
}
else
{
uwSpeedS = uwSpeedS - (1 * 8);
if (uwSpeedS / 8 < 200)
{
ubSimDirS = 1;
}
}
}
//---------------------------------------------------------------------------------------------------
// set engine speed to the requested value
//
else
{
//-------------------------------------------------------------------------------------------
// requested value is greater
//
if (tsECU_ControlS.uwSpeed > uwSpeedS)
{
uwSpeedS = uwSpeedS + 8;
}
//-------------------------------------------------------------------------------------------
// requested value is smaller
//
if (tsECU_ControlS.uwSpeed < uwSpeedS)
{
uwSpeedS = uwSpeedS - 8;
}
}
*pubDataV++ = 0x01;
*pubDataV++ = ubTorqueS; // SPN 512
*pubDataV++ = ubTorqueS; // SPN 513
*pubDataV++ = (uint8_t) uwSpeedS; // SPN 190
*pubDataV++ = (uint8_t) (uwSpeedS >> 8);
*pubDataV++ = tsECU_ControlS.ubDevice; // Controlling device
*pubDataV++ = J1939_UNUSED_UINT8;
*pubDataV++ = J1939_UNUSED_UINT8;
return eJ1939_HANDLER_RESULT_OK;
}
//--------------------------------------------------------------------------------------------------------------------//
// ECU_Pgn65240() //
// Commanded Address //
//--------------------------------------------------------------------------------------------------------------------//
int16_t ECU_Pgn65240(uint8_t * pubDataV, uint16_t uwSizeV, uint8_t CPP_PARM_UNUSED(ubSrcAddrV), uint8_t ubNetV)
{
uint8_t ubNewAddressT;
//----------------------------------------------------------------
// retrieve size information
//
uwSizeV = uwSizeV & J1939_PGN_HANDLER_SIZE_MASK;
//----------------------------------------------------------------
// change address is data size == 9 and NAME is equal
//
if( (J1939_NmtCompareName(ubNetV, pubDataV) == eJ1939_NAME_EQUAL) &&
(uwSizeV == 9) )
{
ubNewAddressT = *(pubDataV + 8);
if(ubNewAddressT < J1939_ADDR_NULL)
{
J1939_MgrInitMsgBuffer(ubNetV, ubNewAddressT);
J1939_NmtClaimAddress(ubNetV);
}
}
return eJ1939_HANDLER_RESULT_OK;
}
//--------------------------------------------------------------------------------------------------------------------//
// ECU_Pgn65242() //
// Software Identification //
//--------------------------------------------------------------------------------------------------------------------//
int16_t ECU_Pgn65242(uint8_t * pubDataV, uint16_t CPP_PARM_UNUSED(uwSizeV), uint8_t CPP_PARM_UNUSED(ubSrcAddrV),
uint8_t ubNetV)
{
uint8_t ubNumOfFieldsT = 2;
uint8_t ubDataCntT;
uint16_t uwSizeT;
//---------------------------------------------------------------------------------------------------
// get the size of the strings
//
uwSizeT = strlen(&szSoftwareVersionT[0]);
uwSizeT = uwSizeT + strlen(&szSoftwareDateT[0]);
uwSizeT = uwSizeT + 1; // the first byte is the number of fields
//---------------------------------------------------------------------------------------------------
// pass size to the stack
//
J1939_PgnSetDataSize(ubNetV, 65242, uwSizeT);
//---------------------------------------------------------------------------------------------------
// copy the data
//
*pubDataV++ = ubNumOfFieldsT;
for(ubDataCntT = 0; ubDataCntT < strlen(&szSoftwareVersionT[0]); ubDataCntT++)
{
*pubDataV++ = szSoftwareVersionT[ubDataCntT];
}
for(ubDataCntT = 0; ubDataCntT < strlen(&szSoftwareDateT[0]); ubDataCntT++)
{
*pubDataV++ = szSoftwareDateT[ubDataCntT];
}
return eJ1939_HANDLER_RESULT_OK;
}
